// ═══════════════════════════════════════════ // مواسم للإنارة — Apps Script Backend v4 // ═══════════════════════════════════════════ var FOLDER_NAME = "مواسم - صور الزيارات"; var SHEET_NAME = "Havells_System"; // ─── الدالة الرئيسية لاستقبال البيانات ─── function doPost(e) { try { var params = JSON.parse(e.postData.contents); // رفع الصورة إلى Drive var photoUrl = ""; if (params.photoBase64 && params.photoBase64.length > 0) { photoUrl = uploadPhoto( params.photoBase64, params.shopName || "محل", params.localTime || new Date().toLocaleString() ); } // حفظ السجل في الشيت var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME) || ss.getActiveSheet(); // إضافة رؤوس الأعمدة إذا كان الشيت فارغاً if (sheet.getLastRow() === 0) { sheet.appendRow([ "وقت الاستلام","المندوب","المحافظة","المنطقة","اسم المحل", "نوع الزيارة","حالة البضاعة","الموقع GPS","الصورة","رابط الصورة", "صاحب المحل","الهاتف","حجم المحل","التصنيف","العميل المرتبط", "ملاحظات العميل","ملاحظات الزيارة","وقت الزيارة" ]); } sheet.appendRow([ new Date(), // A: وقت الاستلام params.agent || "", // B: المندوب params.governorate || "", // C: المحافظة params.area || "", // D: المنطقة params.shopName || "", // E: اسم المحل params.visitType || "", // F: نوع الزيارة params.status || "", // G: حالة البضاعة params.gps || "", // H: الموقع photoUrl ? '=IMAGE("' + photoUrl + '",4,60,80)' : "", // I: صورة مصغرة photoUrl || "", // J: رابط مباشر params.custOwner || "", // K: صاحب المحل params.custPhone || "", // L: الهاتف params.custSize || "", // M: حجم المحل params.custCategory || "", // N: التصنيف params.linkedCustomer|| "", // O: العميل المرتبط params.custNotes || "", // P: ملاحظات العميل params.notes || "", // Q: ملاحظات الزيارة params.localTime || "" // R: وقت الزيارة المحلي ]); return buildResponse({ success: true, photoUrl: photoUrl }); } catch(err) { return buildResponse({ success: false, error: err.message }); } } // ─── رفع الصورة إلى Drive ─── function uploadPhoto(base64, shopName, timeStr) { var folder = getOrCreateFolder(FOLDER_NAME); // تنظيف اسم الملف من الرموز غير المسموحة var safeName = (shopName + "_" + timeStr) .replace(/[\/\\:*?"<>|]/g, "_") .substring(0, 80); var blob = Utilities.newBlob( Utilities.base64Decode(base64), "image/jpeg", safeName + ".jpg" ); var file = folder.createFile(blob); file.setSharing( DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW ); // رابط مباشر يعمل مع =IMAGE() في Sheets return "https://drive.google.com/uc?export=view&id=" + file.getId(); } // ─── الحصول على مجلد أو إنشاؤه ─── function getOrCreateFolder(name) { var it = DriveApp.getFoldersByName(name); return it.hasNext() ? it.next() : DriveApp.createFolder(name); } // ─── بناء الرد مع CORS headers ─── function buildResponse(data) { var output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); return output; } // ─── جلب قائمة العملاء (GET) ─── function doGet(e) { var action = e && e.parameter ? e.parameter.action : ""; if (action === "customers") { return getCustomers(); } return buildResponse({ success: true, message: "Mawasim API v4 ready" }); } function getCustomers() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME) || ss.getActiveSheet(); var data = sheet.getDataRange().getValues(); var customers = []; var seen = {}; for (var i = 1; i < data.length; i++) { var row = data[i]; if (row[5] !== "عميل جديد") continue; var key = (row[4] || "") + "|" + (row[3] || ""); if (seen[key]) continue; seen[key] = true; customers.push({ shopName : row[4] || "", area : row[3] || "", governorate: row[2] || "", owner : row[10] || "", phone : row[11] || "", category : row[13] || "" }); } return buildResponse({ success: true, customers: customers }); } catch(err) { return buildResponse({ success: false, error: err.message }); } }
